import mecanismsData from '../../packages/react-ui/source/data/mecanisms.json'
import ReactMarkdown from 'react-markdown'

export const MechanismsList = ({ children, color }) => (
	<>
		{Object.entries(mecanismsData).map(([mecanismName, meta]) => (
			<div key={mecanismName}>
				<h2>{mecanismName}</h2>
				<ReactMarkdown>{meta.description}</ReactMarkdown>
				{/* Ajouter les exemples et les autres meta-données */}
				{meta.exemples && <ul>
				{Object.entries(meta.exemples ||{}).map(([name, exemple]) => (
						<li key={name}>
							<h3>{name === 'base' ? 'Exemple de base' : name}</h3>
							<ReactMarkdown>{`\`\`\`yaml\n${exemple}\n\`\`\``}</ReactMarkdown>
						</li>
					))}
					</ul>}
			</div>
		))}
	</>
)
